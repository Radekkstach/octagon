import React, { useState, useEffect, useRef } from 'react';
import gsap from 'gsap';

const navLinks = [
  { name: 'Domů', href: '#home' },
  { name: 'O nás', href: '#about' },
  { name: 'Služby', href: '#services' },
  { name: 'Rozvrh', href: '#schedule' },
  { name: 'Kontakt', href: '#contact' },
];

const Navbar = () => {
  const [isScrolled, setIsScrolled] = useState(false);
  const [isMobileMenuOpen, setIsMobileMenuOpen] = useState(false);
  
  // Reference pro animace
  const mobileMenuContainerRef = useRef(null);
  const menuItemsRef = useRef([]);

  // 1. Detekce scrollu (změníme práh na 20px pro rychlejší reakci)
  useEffect(() => {
    const handleScroll = () => {
      setIsScrolled(window.scrollY > 20);
    };
    window.addEventListener('scroll', handleScroll);
    return () => window.removeEventListener('scroll', handleScroll);
  }, []);

  // 2. Animace mobilního menu (Open/Close)
  useEffect(() => {
    if (isMobileMenuOpen) {
      document.body.style.overflow = 'hidden';
      
      // Animace otevření
      const tl = gsap.timeline();
      tl.to(mobileMenuContainerRef.current, { 
        x: '0%', 
        duration: 0.4, 
        ease: 'power3.inOut' 
      })
      .fromTo(menuItemsRef.current,
        { y: 50, opacity: 0 },
        { y: 0, opacity: 1, duration: 0.3, stagger: 0.05, ease: 'back.out(1.7)' },
        "-=0.1"
      );

    } else {
      document.body.style.overflow = 'auto';
      
      // Animace zavření
      gsap.to(mobileMenuContainerRef.current, { 
        x: '100%', 
        duration: 0.4, 
        ease: 'power3.inOut' 
      });
    }
  }, [isMobileMenuOpen]);

  return (
    <>
      {/* HLAVNÍ KONTEJNER NAVBARU 
        - Používáme 'fixed' pro pozici.
        - 'transition-all duration-500' zajistí plynulé "smrsknutí" do pilulky.
      */}
      <nav
        className={`fixed z-50 transition-all duration-500 ease-[cubic-bezier(0.76,0,0.24,1)] flex items-center justify-between
          ${isScrolled 
            ? 'top-4 left-1/2 -translate-x-1/2 w-[90%] md:w-[700px] rounded-full bg-white/5 backdrop-blur-xl border border-white/10 py-3 px-6 shadow-2xl shadow-black/50' 
            : 'top-0 left-0 w-full bg-transparent py-6 px-6 md:px-12 border-b border-transparent'
          }
        `}
      >
        
        {/* LOGO */}
        <a href="#" className="flex items-center gap-1 group">
          {/* Můžeme sem dát SVG logo, zatím text */}
          <span className={`font-heading font-bold uppercase tracking-tighter transition-all duration-300 ${isScrolled ? 'text-xl' : 'text-2xl md:text-3xl'} text-white`}>
            Octagon<span className="text-brand-red group-hover:text-white transition-colors">.</span>
          </span>
        </a>

        {/* DESKTOP LINKS */}
        <div className="hidden md:flex items-center gap-6">
          {navLinks.map((link) => (
            <a
              key={link.name}
              href={link.href}
              className="text-xs font-heading font-bold uppercase tracking-wider text-gray-300 hover:text-white transition-colors duration-200 relative group"
            >
              {link.name}
              {/* Hover podtržítko */}
              <span className="absolute -bottom-1 left-0 w-0 h-[2px] bg-brand-red transition-all duration-300 group-hover:w-full"></span>
            </a>
          ))}
        </div>

        {/* CTA BUTTON / HAMBURGER */}
        <div className="flex items-center gap-4">
          {/* CTA "Permice" - na mobilu schováme, aby se pilulka vešla, nebo necháme jen ikonu */}
          <a 
            href="#membership" 
            className={`hidden md:block bg-brand-red hover:bg-brand-dark text-white font-heading font-bold uppercase rounded-full transition-all duration-300
              ${isScrolled ? 'text-xs py-2 px-5' : 'text-sm py-2 px-6'}
            `}
          >
            Permice
          </a>

          {/* MOBILE HAMBURGER TRIGGER */}
          <button
            onClick={() => setIsMobileMenuOpen(true)}
            className="md:hidden text-white focus:outline-none p-1"
          >
            <svg className="w-6 h-6" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M4 6h16M4 12h16M4 18h16"></path></svg>
          </button>
        </div>
      </nav>

      {/* MOBILE MENU FULLSCREEN OVERLAY 
        - Oddělené od navbaru, aby bylo vždy přes celou obrazovku, i když je navbar "pilulka".
        - Začíná posunuté doprava (translate-x-full), GSAP ho nasune dovnitř.
      */}
      <div 
        ref={mobileMenuContainerRef}
        className="fixed inset-0 bg-gym-base z-[60] flex flex-col translate-x-full" // Výchozí stav mimo obrazovku
      >
        {/* Header mobilního menu */}
        <div className="flex justify-between items-center p-6 border-b border-white/10">
          <span className="text-2xl font-heading font-bold uppercase text-white">Menu</span>
          <button onClick={() => setIsMobileMenuOpen(false)} className="text-white p-2 hover:rotate-90 transition-transform duration-300">
            <svg className="w-8 h-8" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M6 18L18 6M6 6l12 12"></path></svg>
          </button>
        </div>

        {/* Links */}
        <div className="flex-1 flex flex-col justify-center items-center space-y-6">
          {navLinks.map((link, index) => (
            <a
              key={link.name}
              ref={el => menuItemsRef.current[index] = el}
              href={link.href}
              onClick={() => setIsMobileMenuOpen(false)}
              className="text-4xl md:text-5xl font-heading font-bold uppercase text-transparent bg-clip-text bg-gradient-to-r from-gray-400 to-white hover:from-brand-red hover:to-brand-red transition-all duration-300"
            >
              {link.name}
            </a>
          ))}
        </div>

        {/* Footer mobilního menu */}
        <div className="p-8 text-center border-t border-white/10">
           <p className="text-gray-500 font-body text-sm uppercase tracking-widest">Octagon Gym Třebíč</p>
        </div>
      </div>
    </>
  );
};

export default Navbar;
